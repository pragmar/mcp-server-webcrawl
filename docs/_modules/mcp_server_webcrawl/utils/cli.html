def __cli_apply_color(text: str, code: int):
    return f"\033[{code}m{text}\033[0m"

def __cli_light_gray(text):
    return __cli_apply_color(text, "38;2;130;130;130")

def __cli_gold(text):
    return __cli_apply_color(text, "38;2;170;120;0")
